data = [male_survived, male_not_survived, female_survived, female_not_survived]

fig, ax = plt.subplots()

boxes = ax.boxplot(data, labels=['Male Survived', 'Male Not Survived', 'Female Survived', 'Female Not Survived'])

colors = ['#1f77b4', '#ff7f0e', '#2ca02c', '#d62728']
ax.set_title('Age Distribution by Gender and Survival Status')
ax.set_xlabel('Gender and Survival Status')
ax.set_ylabel('Age')
for i, box in enumerate(boxes['boxes']):
    box.set_markerfacecolor(colors[i])

plt.show()

import seaborn as sns

data = pd.DataFrame({'Gender': df['Sex'], 'Age': df['Age'], 'Survived': df['Survived']})

# Set style and context for seaborn
sns.set_style('whitegrid')
sns.set_context('talk')

# Create box plot
ax = sns.boxplot(x='Gender', y='Age', hue='Survived', data=data, palette=['#1f77b4', '#ff7f0e', '#2ca02c', '#d62728'], showfliers=False)

# Set title and axis labels
ax.set_title('Age Distribution by Gender and Survival Status')
ax.set_xlabel('Gender')
ax.set_ylabel('Age')

# Show the plot
plt.show()
